📋 COMMAND REFERENCE
───────────────────────────────────────────────────────────────────────────

Setup:
    python mlbb_complete_setup.py               # Full automated setup
    python manage.py seed_meta                  # Initialize roles/patch
    python manage.py setup_mlbb_tasks            # Setup Celery tasks

Fetch Data:
    python manage.py fetch_mlbb_data --all      # Fetch all heroes
    python manage.py fetch_mlbb_data --stats-only  # Update stats only
    python manage.py fetch_mlbb_data --dry-run  # Preview only

Celery:
    celery -A config worker --beat              # Start all-in-one
    celery -A config worker                     # Start worker only
    celery -A config beat                       # Start scheduler only

Manual:
    python manage.py shell                      # Django console
    python manage.py dbshell                    # Database console
